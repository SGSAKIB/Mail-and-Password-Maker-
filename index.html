import random import string from js import document, navigator, setTimeout # Data first_names = ['John','Jane','Alex','Maria','Chris','Taylor','Jordan','Casey','Morgan','Riley'] last_names = ['Smith','Doe','Brown','Lee','Kim','Patel','Garcia','Wong','Nguyen','Ali','Khan','Chen'] domains = ['gmail.com','outlook.com','yahoo.com','proton.me','icloud.com','zoho.com','mail.com'] def rand_choice(lst): return random.choice(lst) def random_string(length, chars=string.ascii_lowercase + string.digits): return ''.join(random.choice(chars) for _ in range(length)) # Email def make_email(style): if style == 'fake': f = rand_choice(first_names).lower() l = rand_choice(last_names).lower() d = rand_choice(domains) return f"{f}.{l}@{d}" else: return f"{random_string(12)}@tempmail.example" # Password def make_password(length, symbols): length = max(8, int(length)) pools = { 'U': string.ascii_uppercase, 'L': string.ascii_lowercase, 'D': string.digits, 'S': '!@#$%^&*()_+-=[]{}|;:,.<>?' if symbols else '' } chars = pools['U'] + pools['L'] + pools['D'] + pools['S'] pwd = [rand_choice(pools['U']), rand_choice(pools['L']), rand_choice(pools['D'])] if symbols: pwd.append(rand_choice(pools['S'])) while len(pwd) < length: pwd.append(rand_choice(chars)) random.shuffle(pwd) return ''.join(pwd) # Generate def generate(*args): style = document.getElementById('emailType').value length = document.getElementById('passLength').value symbols = document.getElementById('includeSymbols').checked email = make_email(style) pwd = make_password(length, symbols) line = f"{email}:{pwd}" result = document.getElementById('result') result.textContent = line result.style.display = 'block' copy_btn = document.getElementById('copyBtn') copy_btn.style.display = 'block' copy_btn.classList.remove('copied') window.last_line = line #
