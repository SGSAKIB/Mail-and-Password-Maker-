import random import string from js import document, navigator, setTimeout # Data first_names = ['John','Jane','Alex','Maria','Chris','Taylor','Jordan','Casey','Morgan','Riley','Sam','Avery'] last_names = ['Smith','Doe','Brown','Lee','Kim','Patel','Garcia','Wong','Nguyen','Ali','Khan','Chen','Lopez'] domains = ['gmail.com','outlook.com','yahoo.com','proton.me','icloud.com','zoho.com','mail.com','yandex.com'] def rand_choice(lst): return random.choice(lst) def random_string(n): return ''.join(random.choice(string.ascii_lowercase + string.digits) for _ in range(n)) # Email def make_email(style): if style == 'fake': return f"{rand_choice(first_names).lower()}.{rand_choice(last_names).lower()}@{rand_choice(domains)}" else: return f"{random_string(12)}@tempmail.example" # Password def make_password(length, symbols): length = max(8, int(length)) chars = string.ascii_letters + string.digits if symbols: chars += "!@#$%^&*()_+-=[]{}|;:,.<>?" pwd = [ random.choice(string.ascii_uppercase), random.choice(string.ascii_lowercase), random.choice(string.digits) ] if symbols: pwd.append(random.choice("!@#$%^&*()_+-=[]{}|;:,.<>?")) while len(pwd) < length: pwd.append(random.choice(chars)) random.shuffle(pwd) return ''.join(pwd) #
